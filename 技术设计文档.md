## 2. 多模态AI检测系统 🆕

### 2.1 模型架构设计

```python
class MultimodalFusionModel(nn.Module):
    """
    多模态融合模型
    架构参考: FakeSV和SpotFake论文
    """
    def __init__(self, hidden_dim=768, num_classes=3):
        super().__init__()
        
        # 文本编码器 - BERT
        self.text_encoder = BertModel.from_pretrained('hfl/chinese-bert-wwm-ext')
        
        # 视觉编码器 - ResNet50
        self.visual_encoder = ResNet50(pretrained=True)
        self.visual_projection = nn.Linear(2048, hidden_dim)
        
        # 音频编码器 - Whisper + BERT
        self.whisper = whisper.load_model("base")
        
        # 跨模态注意力
        self.cross_modal_attention = nn.MultiheadAttention(
            embed_dim=hidden_dim,
            num_heads=8,
            dropout=0.1
        )
        
        # 融合层
        self.fusion_layer = nn.Sequential(
            nn.Linear(hidden_dim * 3, hidden_dim),
            nn.GELU(),
            nn.Dropout(0.1),
            nn.Linear(hidden_dim, hidden_dim // 2)
        )
        
        # 分类头
        self.classifier = nn.Linear(hidden_dim // 2, num_classes)
```

### 2.2 跨模态注意力机制

```python
class CrossModalAttention(nn.Module):
    """
    跨模态注意力机制
    实现模态间的信息交互和特征增强
    """
    def __init__(self, hidden_dim=768, num_heads=8):
        super().__init__()
        self.hidden_dim = hidden_dim
        self.num_heads = num_heads
        self.head_dim = hidden_dim // num_heads
        
        # Q、K、V投影
        self.query_proj = nn.Linear(hidden_dim, hidden_dim)
        self.key_proj = nn.Linear(hidden_dim, hidden_dim)
        self.value_proj = nn.Linear(hidden_dim, hidden_dim)
        
        # 门控机制
        self.gate = nn.Sequential(
            nn.Linear(hidden_dim * 2, hidden_dim),
            nn.Sigmoid()
        )
    
    def forward(self, query, key, value):
        # 计算注意力权重
        Q = self.query_proj(query)
        K = self.key_proj(key)
        V = self.value_proj(value)
        
        attention_scores = torch.matmul(Q, K.transpose(-2, -1)) / math.sqrt(self.head_dim)
        attention_weights = F.softmax(attention_scores, dim=-1)
        
        # 加权融合
        context = torch.matmul(attention_weights, V)
        
        # 门控融合
        gate_weight = self.gate(torch.cat([query, context], dim=-1))
        output = gate_weight * context + (1 - gate_weight) * query
        
        return output, attention_weights
```

### 2.3 检测流水线

```python
class DetectionPipeline:
    """完整的多模态检测流水线"""
    
    async def detect(self, video_data: bytes, text_data: str = None):
        # 1. 视频帧提取
        frames = self.extract_keyframes(video_data)  # [首、中、尾]
        
        # 2. 音频提取和转写
        audio = self.extract_audio(video_data)
        transcript = await self.whisper.transcribe(audio)
        
        # 3. 文本合并
        full_text = self.merge_text(text_data, transcript)
        
        # 4. 特征提取
        text_features = self.text_encoder(full_text)  # [768]
        visual_features = self.visual_encoder(frames)  # [768]
        audio_features = self.audio_encoder(transcript)  # [768]
        
        # 5. 多模态融合
        fused_features = self.fusion_model(
            text_features, 
            visual_features, 
            audio_features
        )
        
        # 6. 分类预测
        logits = self.classifier(fused_features)
        probabilities = F.softmax(logits, dim=-1)
        
        # 7. 规则增强
        rule_score = self.rule_based_detection(full_text)
        
        # 8. 综合评分
        final_score = self.combine_scores(probabilities, rule_score)
        
        return DetectionResult(
            risk_level=self.get_risk_level(final_score),
            confidence=final_score,
            reasons=self.generate_reasons(text_features, rule_score),
            suggestions=self.generate_suggestions()
        )
```

| App推送 | 中 | 已安装App的家人 |

### 4.2 通知服务架构

```python
class FamilyNotificationService:
    def __init__(self):
        self.rate_limiter = NotificationRateLimiter(
            max_per_hour=5,
            min_interval=300  # 5分钟
        )
    
    async def notify_family(self, event: AlertEvent, contacts: List[FamilyContact]):
        for contact in contacts:
            # 检查频率限制
            if not self.rate_limiter.can_notify(contact.id):
                continue
            
            # 按优先级发送
            for method in contact.notification_preferences:
                if await self._send_notification(method, event, contact):
                    break
```

## 9. 安全和隐私
